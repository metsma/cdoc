cmake_minimum_required(VERSION 3.2)
project(cdoc VERSION 0.0.1)
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED YES)
find_package(LibXml2 REQUIRED)
find_package(OpenSSL REQUIRED)
include_directories(${LIBXML2_INCLUDE_DIR} ${OPENSSL_INCLUDE_DIR})
add_executable(cdoc WIN32 main.cpp Writer.cpp CDOCWriter.cpp CDOCReader.cpp Crypto.cpp Token.cpp)
target_link_libraries(cdoc ${LIBXML2_LIBRARIES} ${OPENSSL_CRYPTO_LIBRARY} ${OPENSSL_SSL_LIBRARY})
